  ```java
  public static <T> void runWorker(java.util.concurrent.Callable<T> bg,
                                   java.util.function.Consumer<T> onSuccess) {
      new SwingWorker<T, Void>() {
          @Override protected T doInBackground() throws Exception { return bg.call(); }
  ```
